### –í–∞—Ä–∏–∞–Ω—Ç 1: –¢–æ–ª—å–∫–æ –±–∞–∑–æ–≤—ã–π —Ä–∞—Å—á—ë—Ç

```python
import math

# –°—Ç—Ä—É–∫—Ç—É—Ä—ã
class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def dist(self, other):
        return math.sqrt((self.x - other.x)**2 + (self.y - other.y)**2)

# –§—É–Ω–∫—Ü–∏—è —Ä–∞—Å—á—ë—Ç–∞
def arc_3points(p1, p2, p3):
    """–í—ã—á–∏—Å–ª–∏—Ç—å —Ü–µ–Ω—Ç—Ä, —Ä–∞–¥–∏—É—Å, —É–≥–ª—ã –¥—É–≥–∏"""
    x1, y1 = p1.x, p1.y
    x2, y2 = p2.x, p2.y
    x3, y3 = p3.x, p3.y
    
    # –¶–µ–Ω—Ç—Ä
    delta = 2 * (x1*(y2-y3) + x2*(y3-y1) + x3*(y1-y2))
    
    if abs(delta) < 1e-10:
        raise ValueError("–¢–æ—á–∫–∏ –∫–æ–ª–ª–∏–Ω–µ–∞—Ä–Ω—ã!")
    
    x1_sq_y1_sq = x1**2 + y1**2
    x2_sq_y2_sq = x2**2 + y2**2
    x3_sq_y3_sq = x3**2 + y3**2
    
    xc = (x1_sq_y1_sq * (y2-y3) + 
          x2_sq_y2_sq * (y3-y1) + 
          x3_sq_y3_sq * (y1-y2)) / delta
    
    yc = (x1_sq_y1_sq * (x3-x2) + 
          x2_sq_y2_sq * (x1-x3) + 
          x3_sq_y3_sq * (x2-x1)) / delta
    
    center = Point(xc, yc)
    radius = p1.dist(center)
    
    # –£–≥–ª—ã
    theta1 = math.atan2(y1 - yc, x1 - xc)
    theta2 = math.atan2(y2 - yc, x2 - xc)
    theta3 = math.atan2(y3 - yc, x3 - xc)
    
    # –ù–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è
    def norm(a):
        while a < 0: a += 2*math.pi
        while a >= 2*math.pi: a -= 2*math.pi
        return a
    
    theta1, theta2, theta3 = norm(theta1), norm(theta2), norm(theta3)
    
    # –í—ã–±–æ—Ä –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è
    if theta1 < theta2 < theta3:
        start_angle, end_angle = theta1, theta3
    elif theta1 < theta3 < theta2:
        start_angle, end_angle = theta3, theta1 + 2*math.pi
    elif theta2 < theta1 < theta3:
        start_angle, end_angle = theta3, theta1 + 2*math.pi
    elif theta2 < theta3 < theta1:
        start_angle, end_angle = theta1, theta3 + 2*math.pi
    elif theta3 < theta1 < theta2:
        start_angle, end_angle = theta1, theta3 + 2*math.pi
    else:
        start_angle, end_angle = theta3, theta1
    
    return {
        'center': center,
        'radius': radius,
        'start_angle': start_angle,
        'end_angle': end_angle
    }

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ

p3 = Point(5, 0)
```

```python
def draw_arc(arc, num_segments=50):
    """–ü–æ–ª—É—á–∏—Ç—å —Ç–æ—á–∫–∏ –¥—É–≥–∏ –¥–ª—è –æ—Ç—Ä–∏—Å–æ–≤–∫–∏"""

    radius = arc['radius']
    start_angle = arc['start_angle']
    end_angle = arc['end_angle']
    
    points = []
    angle_diff = end_angle - start_angle
    
    for i in range(num_segments + 1):
        t = i / num_segments
        angle = start_angle + t * angle_diff
        
        x = center.x + radius * math.cos(angle)
        y = center.y + radius * math.sin(angle)
        
        points.append((x, y))
    
    return points
```
